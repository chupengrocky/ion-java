51      //Boundary writeFloat() -2^128 ~ +2^128
52      //Boundary setIsFloatBinary32Enabled() true & false
53      //when we set setIsFloatBinary32Enabled() as true, we enables writing Binary32 values for floats when there would be no loss in precision.
54      //In this case, we test whether setIsFloatBinary32Enabled() works or not by testing the size of ByteArrayOutputStream.
55      @Test
56      public void testSetIsFloatBinary32Enabled() throws IOException
57      {
58          IonSystem system = IonSystemBuilder.standard().build();
59  
60          IonBinaryWriterBuilder b = IonBinaryWriterBuilder.standard();
61          b.setIsFloatBinary32Enabled(true);
62  
63          ByteArrayOutputStream out = new ByteArrayOutputStream();
64          IonWriter writer = b.build(out);
65          writer.writeFloat(1.5);
66          writer.close();
67          assertEquals(9, out.size());
68          assertEquals(system.newFloat(1.5), system.singleValue(out.toByteArray()));
69  
70          b.setIsFloatBinary32Enabled(false);
71  
72          out = new ByteArrayOutputStream();
73          writer = b.build(out);
74          writer.writeFloat(1.5);
75          writer.close();
76          assertEquals(13, out.size());
77          assertEquals(system.newFloat(1.5), system.singleValue(out.toByteArray()));
78      }

80      //Boundary getSymbolTable().findSymbol()  must be a string
81      //Boundary setInitialSymbolTable() must be a SymbolTable
82      //In this case, we test whether setInitialSymbolTable() can accept a SymbolTable as input, whether getInitialSymbolTable() can return this SymbolTable
83      //and the findSymbol() function can return the right number.
84      //we also test when we call build() multiple times, whether we get not same SymbolTables (but they should have same value).
85  
86      @Test
87      public void testInitialSymtab()
88              throws IOException
89      {
90          SymbolTable sst = _Private_Utils.systemSymtab(1);
91  
92          SymbolTable lst0 = Symtabs.localSymbolTableFactory().newLocalSymtab(sst);
93          lst0.intern("RookieMan");
94  
95          _Private_IonBinaryWriterBuilder b =
96                  _Private_IonBinaryWriterBuilder.standard();
97          b.setInitialSymbolTable(lst0);
98          assertSame(lst0, b.getInitialSymbolTable());
99  
100         OutputStream out = new ByteArrayOutputStream();
101         IonWriter writer = b.build(out);
102         assertEquals(sst.getMaxId() + 1,
103                 writer.getSymbolTable().findSymbol("RookieMan"));
104 
105         // Builder makes a copy of the symtab
106         SymbolTable lst1 = writer.getSymbolTable();
107         assertSame(lst0, b.getInitialSymbolTable());
108 
109         // Second call to build, we get another copy.
110         writer = b.build(out);
111         SymbolTable lst2 = writer.getSymbolTable();
112         writer.writeSymbol("addition");
113 
114         // Now the LST has been extended, so the builder should make a copy
115         // with the original max_id.
116         writer = b.build(out);
117         SymbolTable lst3 = writer.getSymbolTable();
118         assertEquals(sst.getMaxId() + 1,
119                 lst3.findSymbol("RookieMan"));
120 
121 
122         assertEquals(sst.getMaxId() + 1, lst3.getMaxId());
123         assertNotSame(lst0, lst3);
124         assertNotSame(lst1, lst3);
125         assertNotSame(lst2, lst3);
126         assertSame(lst0, b.getInitialSymbolTable());
127     }
