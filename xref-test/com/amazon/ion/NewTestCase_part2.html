17  import org.junit.Test;
18  import com.amazon.ion.system.*;
19  import java.io.*;
20  import static org.junit.Assert.*;

23  public class NewTestCase_part2 extends IonTestCase{
24      @Test
25      public void testFiniteStateMachineReader(){
26          IonReaderBuilder readerBuilder = IonReaderBuilder.standard();
27          String json = "{Key: \"Value\"}";
28          IonReader reader = readerBuilder.build(json);
29          assertEquals(IonType.STRUCT, reader.next());  // position the reader at the first value, a struct
30  
31          reader.stepIn();                              // step into the struct
32          assertEquals(IonType.STRING, reader.next());  // position the reader at the first value in the struct
33  
34          String fieldName = reader.getFieldName();     // retrieve the current value's field name
35          String value = reader.stringValue();          // retrieve the current value's String value
36          assertEquals("Key", fieldName);
37          assertEquals("Value", value);
38          assertNull(reader.next());
39  
40          reader.stepOut();                             // step out of the struct
41          assertNull(reader.next());
42      }
43  
44      @Test
45      public void testFiniteStateMachineText()  throws Exception{
46          ByteArrayOutputStream out = new ByteArrayOutputStream();
47          IonTextWriterBuilder textWriterBuilder = IonTextWriterBuilder.standard();
48  
49  
50          IonWriter textWriter = textWriterBuilder.build(out);
51          textWriter.stepIn(IonType.STRUCT);  // step into a struct
52          textWriter.setFieldName("Key");   // set the field name for the next value to be written
53          textWriter.writeString("Value");    // write the next value
54          textWriter.stepOut();               // step out of the struct
55  
56          assertEquals("{Key:\"Value\"}",out.toString("UTF-8"));
57      }
58  
59      @Test
60      public void testFiniteStateMachineBinary()  throws Exception {
61          IonSystem system = IonSystemBuilder.standard().build();
62          ByteArrayOutputStream out = new ByteArrayOutputStream();
63          IonReaderBuilder readerBuilder = IonReaderBuilder.standard();
64          String json = "{Key: \"Value\"}";
65          IonReader reader = readerBuilder.build(json);
66  
67          IonBinaryWriterBuilder binaryWriterBuilder = IonBinaryWriterBuilder.standard();
68  
69          IonWriter binaryWriter = binaryWriterBuilder.build(out);
70  
71          binaryWriter.writeValues(reader);
72          reader.close();
73          binaryWriter.close();
74  
75          assertEquals("{Key:\"Value\"}", system.singleValue(out.toByteArray()).toString());
76      }
77  }
