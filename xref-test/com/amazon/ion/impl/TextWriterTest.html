44      @Override
45      protected IonWriter makeWriter(OutputStream out, SymbolTable... imports)
46          throws Exception
47      {
48          myOutputForm = OutputForm.TEXT;
49  
50          if (options != null) return options.withImports(imports).build(out);
51  
52          return system().newTextWriter(out, imports);
53      }
54  
55      protected String outputString()
56          throws Exception
57      {
58          byte[] utf8Bytes = outputByteArray();
59          return _Private_Utils.utf8(utf8Bytes);
60      }

98      @Test
99      public void testIvmMinimizing()
100         throws Exception
101     {
102         options = IonTextWriterBuilder.standard();
103         iw = makeWriter();
104         iw.writeSymbol(ION_1_0);
105         iw.writeSymbol(ION_1_0);
106         iw.writeSymbol("foo");
107         iw.writeSymbol(ION_1_0);
108         iw.writeSymbol(ION_1_0);
109         iw.writeSymbol(ION_1_0);
110         iw.writeSymbol("bar");
111         iw.writeSymbol(ION_1_0);
112 
113         String ionText = outputString();
114         assertEquals(ION_1_0 + " " + ION_1_0 + " foo " +
115                      ION_1_0 + " " + ION_1_0 + " " + ION_1_0 + " bar " +
116                      ION_1_0,
117                      ionText);
118 
119         options.setIvmMinimizing(IvmMinimizing.ADJACENT);
120         iw = makeWriter();
121         iw.writeSymbol(ION_1_0);
122         iw.writeSymbol(ION_1_0);
123         iw.writeSymbol("foo");
124         iw.writeSymbol(ION_1_0);
125         iw.writeSymbol(ION_1_0);
126         iw.writeSymbol(ION_1_0);
127         iw.writeSymbol("bar");
128         iw.writeSymbol(ION_1_0);
129 
130         ionText = outputString();
131         assertEquals(ION_1_0 + " foo " + ION_1_0 + " bar " + ION_1_0,
132                      ionText);
133     }

135     @Test
136     public void testLstMinimizing()
137         throws Exception
138     {
139         SymbolTable fred1 = Symtabs.register("fred",   1, catalog());
140 
141         IonBinaryWriter binaryWriter = system().newBinaryWriter(fred1);
142         binaryWriter.writeSymbol("fred_1");
143         binaryWriter.writeSymbol("ginger");
144         binaryWriter.finish();
145         byte[] binaryData = binaryWriter.getBytes();
146 
147         options = IonTextWriterBuilder.standard();
148 
149         // TODO User reader still transfers local symtabs!
150         IonReader binaryReader = system().newReader(binaryData);
151         iw = makeWriter();
152         iw.writeValues(binaryReader);
153 
154         String ionText = outputString();
155         assertEquals(// TODO "$ion_1_0 " +
156                      "$ion_symbol_table::{imports:[{name:\"fred\",version:1,max_id:2}],"
157                      +                   "symbols:[\"ginger\"]} " +
158                      "fred_1 ginger",
159                      ionText);
160 
161         options.setLstMinimizing(LstMinimizing.LOCALS);
162         binaryReader = system().newReader(binaryData);
163         iw = makeWriter();
164         iw.writeValues(binaryReader);
165         ionText = outputString();
166         assertEquals(// TODO "$ion_1_0 " +
167                      "$ion_symbol_table::{imports:[{name:\"fred\",version:1,max_id:2}]} " +
168                      "fred_1 ginger",
169                      ionText);
170 
171         options.setLstMinimizing(LstMinimizing.EVERYTHING);
172         binaryReader = system().newReader(binaryData);
173         iw = makeWriter();
174         iw.writeValues(binaryReader);
175         ionText = outputString();
176         assertEquals(// TODO "$ion_1_0 " +
177                      "$ion_1_0 fred_1 ginger",
178                      ionText);
179 
180         options.setInitialIvmHandling(SUPPRESS);
181         binaryReader = system().newReader(binaryData);
182         iw = makeWriter();
183         iw.writeValues(binaryReader);
184         ionText = outputString();
185         assertEquals("fred_1 ginger",
186                      ionText);
187     }

216     @Test
217     public void testWritingLongStrings()
218         throws Exception
219     {
220         options = IonTextWriterBuilder.standard();
221         options.setInitialIvmHandling(SUPPRESS);
222         options.setLongStringThreshold(5);
223 
224         // This is tricky because we must avoid triple-quoting multiple
225         // long strings in such a way that they'd get concatenated together!
226         IonDatagram dg = system().newDatagram();
227         dg.add().newNullString();
228         dg.add().newString("hello");
229         dg.add().newString("hello\nnurse");
230         dg.add().newString("goodbye").addTypeAnnotation("a");
231         dg.add().newString("what's\nup\ndoc");
232 
233         expectRendering("null.string \"hello\" '''hello\nnurse''' a::'''goodbye''' \"what's\\nup\\ndoc\"",
234                         dg);
235 
236         dg.clear();
237         dg.add().newString("looong");
238         IonSequence seq = dg.add().newEmptySexp();
239         seq.add().newString("looong");
240         seq.add().newString("hello");
241         seq.add().newString("hello\nnurse");
242         seq.add().newString("goodbye").addTypeAnnotation("a");
243         seq.add().newString("what's\nup\ndoc");
244 
245         expectRendering("'''looong''' ('''looong''' \"hello\" '''hello\nnurse''' a::'''goodbye''' \"what's\\nup\\ndoc\")",
246                         dg);
247 
248         dg.clear();
249         dg.add().newString("looong");
250         seq = dg.add().newEmptyList();
251         seq.add().newString("looong");
252         seq.add().newString("hello");
253         seq.add().newString("hello\nnurse");
254         seq.add().newString("what's\nup\ndoc");
255 
256         expectRendering("'''looong''' ['''looong''',\"hello\",'''hello\n" +
257                         "nurse''','''what\\'s\n" +
258                         "up\n" +
259                         "doc''']",
260                         dg);
261 
262         options.setLongStringThreshold(0);
263         expectRendering("\"looong\" [\"looong\",\"hello\",\"hello\\nnurse\",\"what's\\nup\\ndoc\"]",
264             dg);
265 
266         options.setLongStringThreshold(5);
267         dg.clear();
268         dg.add().newString("looong");
269         IonStruct struct = dg.add().newEmptyStruct();
270         struct.add("a").newString("looong");
271         struct.add("b").newString("hello");
272         struct.add("c").newString("hello\nnurse");
273         struct.add("d").newString("what's\nup\ndoc");
274 
275         expectRendering(
276             "'''looong''' {a:'''looong''',b:\"hello\",c:'''hello\n" +
277             "nurse''',d:'''what\\'s\n" +
278             "up\n" +
279             "doc'''}",
280             dg
281         );
282 
283         options.withPrettyPrinting();
284         expectRendering(
285             // TODO amzn/ion-java/issues/57 determine if these really should be platform independent newlines
286             format(
287                 "%n" +
288                 "'''looong'''%n" +
289                 "{%n" +
290                         "  a:'''looong''',%n" +
291                 "  b:\"hello\",%n" +
292                 "  c:'''hello\n" +
293                         "nurse''',%n" +
294                         "  d:'''what\\'s\n" +
295                         "up\n" +
296                         "doc'''%n" +
297                         "}"
298             ),
299             dg
300         );
301     }

358     @Test
359     public void testWritingJsonLongClobs()
360         throws Exception
361     {
362         options = IonTextWriterBuilder.json();
363         options.setInitialIvmHandling(SUPPRESS);
364         options.setLongStringThreshold(4);
365 
366         // Cannot call expectRendering as the input to reload() would be Json rather than Ion
367         // test 1
368         IonDatagram dg = system().newDatagram();
369         dg.add().newClob(new byte[]{'a', '"', '\'', '\n'});
370 
371         // expectRendering
372         iw = makeWriter();
373         dg.writeTo(iw);
374 
375         IonDatagram reloaded = loader().load("{{\"a\\\"'\\n\"}}");
376         assertEquals(dg, reloaded);
377 
378         String actual = outputString();
379         assertEquals("\"a\\\"'\\n\"", actual);
380 
381         // test 2
382         dg.clear();
383         dg.add().newClob(new byte[]{'a', '"', '\'', '\n', 'c', 0x7F});
384 
385         // expectRendering
386         iw = makeWriter();
387         dg.writeTo(iw);
388 
389         reloaded = loader().load("{{'''a\"\\'\nc\\x7f'''}}");
390         assertEquals(dg, reloaded);
391 
392         actual = outputString();
393         assertEquals("\"a\\\"'\\nc\\u007f\"", actual);
394     }
